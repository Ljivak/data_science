to make new git repository: 
git init 
git add . 
git commit -m "initial commit" 
git branch -M main 
git remote add origin git@github.com:Ljivak/data_science.git
git push -u origin main
echo".DS_Store" > .gitignore  // for deleting unnecessary macos file
git add .gitignore
git commit -m "Ignore .DS_Store"
git push

to push in already existing repository:
git add . 
git commit -m "description" 
git push

git restore file -- makes your file as last version in git
git reset HEAD -- deletes files from staging area 
git reset HEAD~1 -- deletes last commit

git revert <hash> -- reverts commit under hash and commits automatically (-n to put in st. area)
